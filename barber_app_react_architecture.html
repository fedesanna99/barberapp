hooks/useAuth.ts
export function useAuth() {
  const [session, setSession] = useState(null)
  const [profile, setProfile] = useState(null)

  useEffect(() => {
    supabase.auth.getSession().then(({ data }) => {
      setSession(data.session)
      if (data.session) fetchProfile(data.session.user.id)
    })

    const { data: listener } = supabase.auth.onAuthStateChange(
      (_event, session) => {
        setSession(session)
        if (session) fetchProfile(session.user.id)
      }
    )
    return () => listener.subscription.unsubscribe()
  }, [])

  async function fetchProfile(userId: string) {
    const { data } = await supabase
      .from('profiles')
      .select('*, barbers(*)')
      .eq('id', userId)
      .single()
    setProfile(data)
  }

  const signInWithGoogle = () =>
    supabase.auth.signInWithOAuth({ provider: 'google' })

  const signOut = () => supabase.auth.signOut()

  return { session, profile, signInWithGoogle, signOut }
}
The select('*, barbers(*)') join means one query tells you both the user's profile AND whether they are a barber — no second round-trip needed.
hooks/useFeed.ts — paginated feed from followed barbers
export function useFeed(userId: string) {
  const [posts, setPosts] = useState([])
  const [page, setPage] = useState(0)
  const PAGE = 12

  useEffect(() => {
    supabase
      .from('posts')
      .select(`
        *,
        barbers ( id, profile:profiles(display_name, avatar_url) )
      `)
      .in(
        'barber_id',
        supabase
          .from('follows')
          .select('barber_id')
          .eq('follower_id', userId)
      )
      .order('created_at', { ascending: false })
      .range(page * PAGE, (page + 1) * PAGE - 1)
      .then(({ data }) => setPosts(prev => [...prev, ...(data ?? [])]))
  }, [userId, page])

  return { posts, loadMore: () => setPage(p => p + 1) }
}
hooks/useBarbers.ts — discover with sort + geo
type Sort = 'nearby' | 'popular' | 'new'

export function useBarbers(sort: Sort, userLat?: number, userLng?: number) {
  const [barbers, setBarbers] = useState([])

  useEffect(() => {
    let q = supabase
      .from('barbers')
      .select('*, profile:profiles(display_name, avatar_url, lat, lng)')

    if (sort === 'popular') q = q.order('rating', { ascending: false })
    if (sort === 'new')     q = q.order('created_at', { ascending: false })

    q.then(({ data }) => {
      let result = data ?? []
      if (sort === 'nearby' && userLat && userLng) {
        result = result.sort((a, b) =>
          distance(a.profile, userLat, userLng) -
          distance(b.profile, userLat, userLng)
        )
      }
      setBarbers(result)
    })
  }, [sort])

  return { barbers }
}

function distance(p, lat, lng) {
  const R = 6371
  const dLat = (p.lat - lat) * Math.PI / 180
  const dLng = (p.lng - lng) * Math.PI / 180
  const a = Math.sin(dLat/2)**2 +
    Math.cos(lat*Math.PI/180) * Math.cos(p.lat*Math.PI/180) *
    Math.sin(dLng/2)**2
  return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a))
}
For v1, the haversine distance sort runs client-side — fast enough for <500 barbers. When you scale past that, move it to a Postgres function using the PostGIS extension Supabase ships with.
hooks/useAvailability.ts — compute open slots
export function useAvailability(barberId: string, date: Date) {
  const [slots, setSlots] = useState<string[]>([])
  const [booked, setBooked] = useState<string[]>([])

  useEffect(() => {
    const dow = date.getDay()           // 0 = Sun … 6 = Sat
    const dateStr = date.toISOString().split('T')[0]

    Promise.all([
      supabase
        .from('availability')
        .select('start_time, end_time')
        .eq('barber_id', barberId)
        .eq('day_of_week', dow),
      supabase
        .from('bookings')
        .select('time_slot')
        .eq('barber_id', barberId)
        .eq('date', dateStr)
        .neq('status', 'cancelled')
    ]).then(([avail, existing]) => {
      const window = avail.data?.[0]
      if (!window) return setSlots([])

      const all = generateSlots(window.start_time, window.end_time)
      const taken = new Set(existing.data?.map(b => b.time_slot))
      setSlots(all)
      setBooked(all.filter(s => taken.has(s)))
    })
  }, [barberId, date.toDateString()])

  return { slots, booked }
}

function generateSlots(start: string, end: string, step = 30) {
  const slots = []
  let [h, m] = start.split(':').map(Number)
  const [eh, em] = end.split(':').map(Number)
  while (h * 60 + m < eh * 60 + em) {
    slots.push(`${String(h).padStart(2,'0')}:${String(m).padStart(2,'0')}`)
    m += step
    if (m >= 60) { h++; m -= 60 }
  }
  return slots
}
hooks/useBooking.ts — create a booking
export function useBooking() {
  const [loading, setLoading] = useState(false)

  async function createBooking({ clientId, barberId, date, timeSlot }) {
    setLoading(true)
    const { data, error } = await supabase
      .from('bookings')
      .insert({
        client_id: clientId,
        barber_id: barberId,
        date: date.toISOString().split('T')[0],
        time_slot: timeSlot,
        status: 'pending'
      })
      .select()
      .single()
    setLoading(false)
    return { data, error }
  }

  async function cancelBooking(bookingId: string) {
    return supabase
      .from('bookings')
      .update({ status: 'cancelled' })
      .eq('id', bookingId)
  }

  return { createBooking, cancelBooking, loading }
}
Status flow: pending → confirmed (barber taps confirm) → done (auto after appointment) or cancelled. The barber's dashboard listens to the bookings table via Supabase Realtime to get instant alerts.
Row-Level Security — paste these into the Supabase SQL editor
-- POSTS: anyone can read, only owning barber can write
create policy "posts_select" on posts
  for select using (true);

create policy "posts_insert" on posts
  for insert with check (
    barber_id = (
      select id from barbers where profile_id = auth.uid()
    )
  );

-- BOOKINGS: clients see own rows, barbers see incoming
create policy "bookings_client" on bookings
  for select using (client_id = auth.uid());

create policy "bookings_barber" on bookings
  for select using (
    barber_id = (
      select id from barbers where profile_id = auth.uid()
    )
  );

create policy "bookings_insert" on bookings
  for insert with check (client_id = auth.uid());

create policy "bookings_cancel" on bookings
  for update using (
    client_id = auth.uid() or
    barber_id = (
      select id from barbers where profile_id = auth.uid()
    )
  );

-- AVAILABILITY: only the owning barber can manage
create policy "availability_select" on availability
  for select using (true);

create policy "availability_write" on availability
  for all using (
    barber_id = (
      select id from barbers where profile_id = auth.uid()
    )
  );
Enable RLS on every table first: alter table bookings enable row level security; — Supabase defaults to deny-all once RLS is on, so the policies above are additive grants, not overrides.
